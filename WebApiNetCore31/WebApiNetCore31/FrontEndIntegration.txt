 //Service
 getWithEtag(id: number): Observable<any> {
        return this.httpClient.get(`https://localhost:5001/api/RestApi/${id}`, {
            headers: {
                'Access-Control-Expose-Headers': "Etag",
                'Content-Type': 'text/plain; charset=utf-8',
                'If-None-Match': this.etag
            },
            observe: 'response',
            responseType: 'text'
        })
    }
